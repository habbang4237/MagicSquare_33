### `src/boundary/input_contract_validator.py`

```python
"""FR-01 size validation (AC-FR-01-01 only)."""

from __future__ import annotations

from typing import Any

from boundary.error_messages import INVALID_SIZE_CODE, INVALID_SIZE_MESSAGE
from boundary.error_response import ErrorResponse

_GRID_DIMENSION: int = 4

class InputContractValidator:
    """Validates 4×4 grid structure; size failures only for this slice."""

    def validate(self, grid: Any) -> ErrorResponse | None:
        """Return ``ErrorResponse`` on size violation, else ``None``."""
        if self._is_invalid_size(grid):
            return ErrorResponse(
                code=INVALID_SIZE_CODE,
                message=INVALID_SIZE_MESSAGE,
            )
        return None

    @staticmethod
    def _is_invalid_size(grid: Any) -> bool:
        if grid is None or not isinstance(grid, list):
            return True
        if len(grid) != _GRID_DIMENSION:
            return True
        for row in grid:
            if not isinstance(row, list) or len(row) != _GRID_DIMENSION:
                return True
        return False
```

### `src/boundary/puzzle_boundary.py`

```python
"""Boundary entry: input contract then use-case delegation."""

from __future__ import annotations

from typing import Any

from boundary.error_response import ErrorResponse
from boundary.input_contract_validator import InputContractValidator
from control.solve_puzzle_use_case import SolvePuzzleUseCase

class PuzzleBoundary:
    """Receives external grid; rejects invalid size before control."""

    def __init__(self, use_case: SolvePuzzleUseCase) -> None:
        self._use_case = use_case
        self._validator = InputContractValidator()

    def receive(self, grid: Any) -> ErrorResponse | list[int]:
        """Validate size; on failure return ``ErrorResponse`` without calling use case."""
        error = self._validator.validate(grid)
        if error is not None:
            return error
        return self._use_case.execute(grid)
```
